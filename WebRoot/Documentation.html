package com.mycomp.gui.sample.pages;

import com.endress.infoserve.gui.Page;
import com.endress.infoserve.gui.form.Label;

public class HelloWorldSample extends Page{
    
    protected void pageInit() {
        
        addHeaderCSS("./wgt-resources/guicontrol.css");
        addHeaderCSS("./wgt-resources/guicontrol_bb.css");

        // add label
        Label label1 = getFactory().createLabel("Hello Web!");
        add( label1 );
    }

    protected void pageLoad() {     
    }

    protected String title() {
        return "Hello Web Sample";
    }
}
